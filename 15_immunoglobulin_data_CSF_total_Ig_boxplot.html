def contrastMWU( data , groupCol , testCol , contrasts = contrasts):

    interLevelScale= 0.01

    pvalL = []
    for contrast in contrasts:    
        mC1 = data[groupCol] == contrast[0]
        mC2 = data[groupCol] == contrast[1]

        nN = ~data[testCol].isna()
        X1 = data.loc[ nN&mC1 , testCol ]
        X2 = data.loc[ nN&mC2 , testCol ]

        s,pval = stats.mannwhitneyu(X1,X2, method='exact') 

        pvalL.append( pval )



    reject , pval_corrected , alpha1, aplha2 = multipletests( pvalL , 
                                                             alpha = 0.05 , method = 'fdr_bh' , is_sorted=False) 
    MWUpval_cor = { c:pval_corrected[i] for i,c in enumerate(contrasts) }
    return MWUpval_cor

def plotContrasts( data , testCol , MWUpval_cor , ax , WH , group2position ,logScale=False , 
                  contrasts = contrasts , level = 1,interLevelScale= 0.2):
    
    ## create the bracket
    m = data[testCol].min()
    M = data[testCol].max()

    for i,contrast in enumerate(contrasts):
        

        ## get significance:
        padj = MWUpval_cor[ contrast ]
        if padj >=0.05:
            continue
            
        symbol = 'NS'
        bracketColor = 'grey'
        if padj <0.05:
            bracketColor='black'
            if padj>=0.01:
                symbol = '*'
            else:
                symbol = '*' * -int( np.log10(padj) )
        #print(np.log10(padj) , symbol)


        x1 = group2position[ contrast[0] ]
        x2 = group2position[ contrast[1] ]

        #print(contrast , x1 , x2)
        step = max( (M-m) * 14/1000 * WH*0.95  , interLevelScale * M)
        print(step)
        y = M + (step * level )
        yp = M + (step * (level - 0.25) )
        if logScale:
            
            #step = max( (M/m) * 14/1000 * WH*0.95  , interLevelScale * M)
            step= np.log10(  M/max(m,0.1))*0.5 # * 14/1000 * WH*0.95 
            print(step)
            y = M * ((step ** level[contrast] ))
            yp = M * ((step ** (level[contrast] - 0.25) ))
        #print( x1,x2 , y )

        ax.plot( [x1,x1,x2,x2] , [yp,y,y,yp] , color=bracketColor )
        ax.text( (x1+x2)*0.5 , y , symbol , font ='monospace' , fontsize = 14 , color = bracketColor)
        level+=1
    if level == 1: ## no test significant
        ax.text( (min( group2position.values() ) + max( group2position.values() )) / 2  , M + (M-m)*0.05, 
                'All Mann-Whitney U\ntest non-significant', 
                   fontname = 'Arial',
                   color='grey', fontsize=15,
               horizontalalignment='center')
        ax.set_ylim(m - (M-m)*0.1,M + (M-m)*0.2)
