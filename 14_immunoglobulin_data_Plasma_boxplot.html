def contrastMWU( data , groupCol , testCol , contrasts = contrasts):
    """ performs exact MWU test for a set of pairs of groups, 
    corrects pvalues using BH procedure.
    Report results as dictionnary """
    
    resD = {'contrast':[],
        'pval':[],
        'N1':[],
        'N2':[],
        'Median1':[],
        'Median2':[]}
    
    for contrast in contrasts:    
        mC1 = data[groupCol] == contrast[0]
        mC2 = data[groupCol] == contrast[1]

        nN = ~data[testCol].isna()
        X1 = data.loc[ nN&mC1 , testCol ]
        X2 = data.loc[ nN&mC2 , testCol ]

        s,pval = stats.mannwhitneyu(X1,X2, method='exact') 

        resD['contrast'].append( contrast )
        resD['pval'].append( pval )
        resD['N1'].append( len(X1) )
        resD['N2'].append( len(X2) )
        resD['Median1'].append( X1.median() )
        resD['Median2'].append( X2.median() )

    reject , pval_corrected , alpha1, aplha2 = multipletests( resD['pval'] , 
                                                             alpha = 0.05 , method = 'fdr_bh' , is_sorted=False) 
    
    resD['padj'] = pval_corrected
    return resD

def makeMolecPanel( data , molecName , ax , WH , ylabSize=20 , xtick_fontsize=16 , title_size=25  ):
    """function specific to the immunoglobulin plots.
        Returns a dictionnary with the MWU test results"""
    
    ## special colors for specific points
    data['color'] = 'grey'
    if molecName.startswith('CSF_'):
        data.loc[ ['NC015','NC023','NC040'] ,'color'] = 'red'
        if molecName.endswith('IgG'):
            data.loc[ ['NC039 (ZH5)'] ,'color'] = 'orange'
        elif molecName.endswith('IgA'):
            data.loc[ ['NC019','NC026','NC039 (ZH5)'] ,'color'] = 'orange'


    #ax.axhline(0 , color='grey')
    sns.boxplot( x="Stage" , y = molecName , data=data , 
                     order = order5 , palette = palette5 , ax =ax ,  showfliers = False )
    sns.stripplot( x="Stage" , y = molecName , data=data , 
                     order = order5 , hue= "color" , 
                  palette = ['grey','red','orange'] , hue_order=['grey','red','orange'] , 
                  ax =ax , size= 10*WH/10)

    ax.legend([],[], frameon=False)
    ax.set_xlabel( 'class' ,fontname = 'Arial' , size = ylabSize)
    ax.set_ylabel( 'ΔOD450' ,fontname = 'Arial' , size = ylabSize)
    ax.set_xticks([group2position[x] for x in order5],
                  [x.replace(' ','\n') for x in order5], 
                  fontname = 'Arial' , fontsize= xtick_fontsize )
    ax.set_title( m2title[molecName] , fontname = 'Arial' , size=title_size)

    ax.xaxis.grid() # vertical lines
    interLevelScale= 0.01

    MWU_results = contrastMWU( data , groupCol='Stage' , testCol=molecName , contrasts = contrasts)

    MWUpval_cor = { MWU_results['contrast'][i]:MWU_results['padj'][i] for i in range(len(MWU_results['contrast'])) }

    ## create the bracket
    m = data[molecName].min()
    M = data[molecName].max()

    level=1
    for i,contrast in enumerate(contrasts):

        if MWUpval_cor[ contrast ]>=0.05:
            continue ## ignore non-sig tests

        ## get significance:
        padj = MWUpval_cor[ contrast ]
        symbol = 'NS'
        bracketColor = 'grey'
        if padj <0.05:
            bracketColor='black'
            if padj>=0.01:
                symbol = '*'
            else:
                symbol = '*' * -int( np.log10(padj) )

        x1 = group2position[ contrast[0] ]
        x2 = group2position[ contrast[1] ]

        
        step = max( (M-m) * 14/1000 * WH*0.95  , interLevelScale * M)
        #print(step)
        y = M + (step * level )
        yp = M + (step * (level - 0.25) )
        #print( x1,x2 , y )

        ax.plot( [x1,x1,x2,x2] , [yp,y,y,yp] , color=bracketColor )
        ax.text( (x1+x2)*0.5 , y , symbol , ha='center',
                font ='monospace' , fontsize = 14 , color = bracketColor)
        level += 1
    if level == 1: ## no test significant
        ax.text(2, M + (M-m)*0.05, 'All Mann-Whitney U test non-significant', ha='center',
            
                   fontname = 'Arial',
                   color='grey', fontsize=15)
        ax.set_ylim(m - (M-m)*0.1,M + (M-m)*0.2)

    return MWU_results
