The plots obtained using marginalized data appear in Figure 3 C.

for i,m in enumerate( inflammatoryMolecules + neuroMolecules ):

    fig,ax = plt.subplots(figsize=(7,7))
    
    Tau = {}
    for g in categories:
            
        c1 = df_molecular.loc[ indices[g] , 'CSF_'+m]
        c2 = df_molecular.loc[ indices[g] , 'Plasma_'+m]
        t,p = stats.kendalltau(list(c1),list(c2) , nan_policy = 'omit')
    
        Tau[g] = t
    
        if g != 'all':
            ax.scatter( c1,c2 , c=palette5[g] , label=g.ljust(20)+':{:.2f}'.format(t) )
        else:
            ax.scatter( [],[] , c='white' , label=g.ljust(20)+':{:.2f}'.format(t) )
    
    fM1 = df_molec_md.loc[ 'CSF_'+m , 'fractionMissing']
    fM2 = df_molec_md.loc[ 'Plasma_'+m , 'fractionMissing']
        
        
    if m =='OPG':
        m = "TNFRSF11B"

    ax.set_xlabel('CSF_'+m + " - missing fraction: {:.2f}".format(fM1))
    ax.set_ylabel('Plasma_'+m + " - missing fraction: {:.2f}".format(fM2))
    
    leg = ax.legend(title="Kendall's Tau")
    leg._legend_box.align = "right"
    
    fN = "images/correlation/correlation_{}.pdf".format(m)
    
    fig.savefig(fN)
    plt.close()

import matplotlib.transforms as transforms

def confidence_ellipse(x, y, ax, n_std=3.0, facecolor='none', **kwargs):
    """
    Create a plot of the covariance confidence ellipse of *x* and *y*.

    Parameters
    ----------
    x, y : array-like, shape (n, )
        Input data.

    ax : matplotlib.axes.Axes
        The axes object to draw the ellipse into.

    n_std : float
        The number of standard deviations to determine the ellipse's radiuses.

    **kwargs
        Forwarded to `~matplotlib.patches.Ellipse`

    Returns
    -------
    matplotlib.patches.Ellipse
    """
    if x.size != y.size:
        raise ValueError("x and y must be the same size")

    cov = np.cov(x, y)
    #print(cov)
    if ( cov[0, 0] * cov[1, 1] ) == 0:
        if cov[0, 0]  == cov[1, 1] :
            ## variance is 0 is both dimemsion:
            return None
        ## variance is 0 is at least 1 dimension
        ell_radius_x = cov[0,0]
        ell_radius_y = cov[1,1]
    else:
        pearson = max( -1.0 ,min( cov[0, 1]/np.sqrt(cov[0, 0] * cov[1, 1]) , 1.0))
        # Using a special case to obtain the eigenvalues of this
        # two-dimensionl dataset.
        ell_radius_x = np.sqrt(1 + pearson)
        ell_radius_y = np.sqrt(1 - pearson)
    ellipse = Ellipse((0, 0), width=ell_radius_x * 2, height=ell_radius_y * 2,
                      facecolor=facecolor, **kwargs)

    # Calculating the stdandard deviation of x from
    # the squareroot of the variance and multiplying
    # with the given number of standard deviations.
    scale_x = np.sqrt(cov[0, 0]) * n_std
    mean_x = np.mean(x)

    # calculating the stdandard deviation of y ...
    scale_y = np.sqrt(cov[1, 1]) * n_std
    mean_y = np.mean(y)

    transf = transforms.Affine2D() \
        .rotate_deg(45) \
        .scale(scale_x, scale_y) \
        .translate(mean_x, mean_y)

    ellipse.set_transform(transf + ax.transData)
    return ellipse
